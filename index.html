function drawWheel(playersData) { let ctx = document.getElementById('wheel').getContext('2d'); let totalBet = playersData.reduce((a, b) => a + b.bet, 0); let startAngle = 0; playersData.forEach(player => { let sliceAngle = (player.bet / totalBet) * 2 * Math.PI; ctx.fillStyle = player.color; ctx.beginPath(); ctx.moveTo(250, 250); ctx.arc(250, 250, 250, startAngle, startAngle + sliceAngle); ctx.fill(); startAngle += sliceAngle; }); }
